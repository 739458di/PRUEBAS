function detectSellerLead(texto, rawMsg) {
    if (!texto) return false;
    var t = texto.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
    if (t.length < 3) return false;

    // If message came from a CAR LISTING ad (has price/$, car brand, Fyradrive listing) → BUYER, not seller
    var adCombined = '';
    if (rawMsg && rawMsg.message) {
        var _extMsg = rawMsg.message.extendedTextMessage;
        if (_extMsg) {
            adCombined = ((_extMsg.title || '') + ' ' + (_extMsg.description || '') + ' ' + (_extMsg.canonicalUrl || '')).toLowerCase();
        }
        // Also check contextInfo for external ad reply (Click-to-WhatsApp ads)

        if (_ctxInfo && _ctxInfo.externalAdReply) {
            var _adR = _ctxInfo.externalAdReply;
            adCombined += ' ' + ((_adR.title || '') + ' ' + (_adR.body || '') + ' ' + (_adR.description || '') + ' ' + (_adR.sourceUrl || '')).toLowerCase();
        }
    }
    // Also check [DESC:] and [ANUNCIO:] tags in texto
    var descMatch = t.match(/\[desc:\s*([^\]]+)\]/);
    var anuncioMatch = t.match(/\[anuncio:\s*([^\]]+)\]/);
    if (descMatch) adCombined += ' ' + descMatch[1];
    if (anuncioMatch) adCombined += ' ' + anuncioMatch[1];

    // CAR LISTING detection: if ad has price + car info → this is a BUYER responding to a listing
    var adHasPrice = /\$\s*[\d,.]+|\d{2,3},?\d{3}/.test(adCombined);
    var adHasCarInfo = /(chrysler|honda|toyota|ford|chevrolet|chevy|nissan|mazda|vw|volkswagen|hyundai|kia|bmw|dodge|jeep|ram|suzuki|mitsubishi|audi|mercedes|gmc|buick|lincoln|cadillac|subaru|acura|infiniti|lexus|renault|peugeot|seat|fiat|mini|mg|changan|chirey|omoda|tiggo|cupra|chery)/.test(adCombined);

    }

    if (adIsSellAd) {
